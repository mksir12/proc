  <script>
    function parseUrls(raw) {
      return raw.split(/[\n,]+/g)
        .map(s => s.trim())
        .map(s => s.replace(/^"+|"+$/g, '')) // remove wrapping quotes
        .map(s => s.replace(/,+$/g, '')) // remove trailing commas
        .filter(s => s && /^https?:\/\//.test(s));
    }

    document.getElementById('submit').addEventListener('click', () => {
      let urls = parseUrls(document.getElementById('urls').value || "");
      if (!urls.length) {
        alert("Please paste at least one valid URL.");
        return;
      }
      const asJson = document.getElementById('asJson').checked;
      const urlParam = urls.map(encodeURIComponent).join(',');
      let target = `/api/check?url=${urlParam}`;
      if (asJson) target += "&format=json";
      window.location.href = target; // redirect to API result
    });
  </script>
